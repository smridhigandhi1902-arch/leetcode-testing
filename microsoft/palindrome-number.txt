class Solution {
public:
    bool isPalindrome(int x) {
        if (x<0 ) return false;
        int reversenum=0;
        int original = x;
        while(x != 0){
           int lastdigit = x % 10;
            if( reversenum > INT_MAX/10 || reversenum < INT_MIN/10){
                return false;
            }
            reversenum = (reversenum*10) + lastdigit;
            x= x/10;
        }
       
            return original == reversenum;
        
    }
};
